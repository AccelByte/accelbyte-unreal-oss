84template <typename T>
85class TOnlineSubsystemAccelByteConfigEntry
86{
87// Types
88public:
89 using FValidatorFunction = TFunction<bool(const T&)>;
90
91public:
96 bool Load()
97 {
98 if (Section.IsEmpty())
99 {
100 return false;
101 }
102
103 if (Key.IsEmpty())
104 {
105 return false;
106 }
107
108 if (TIsTArray<T>::Value)
109 {
110 TArray<FString> StringArray{};
111
112 // #NOTE: LoadABConfigFallback does not support arrays, and we do not have a format for arrays on the command
113 // line in place. For now, just directly grab the array from the config. In the future, we may want to convert
114 // this to allow for command line overriding.
115 const int32 ElemNum = GConfig->GetArray(*Section
116 , *Key
117 , StringArray
118 , GEngineIni);
119
120 if (ElemNum <= 0)
121 {
122 return false;
123 }
124
125 return SetValueFromStringArray(StringArray);
126 }
127 else
128 {
129 FString ConfigValueStr{};
130 const bool bExists = FAccelByteUtilities::LoadABConfigFallback(Section
131 , Key
132 , ConfigValueStr);
133
134 if (!bExists)
135 {
136 return false;
137 }
138
139 return SetValueFromString(ConfigValueStr);
140 }
141 }
142
147 void SetValue(const T& InValue)
148 {
149 if (Validator && !Validator(InValue))
150 {
151 return;
152 }
153
154 Value = InValue;
155 }
156
161 void SetValue(T&& InValue)
162 {
163 if (Validator && !Validator(InValue))
164 {
165 return;
166 }
167
168 Value = InValue;
169 }
170
174 bool SetValueFromString(const FString& InStringValue)
175 {
176 TOnlineSubsystemAccelByteConfigValueConverter<T> Converter{};
177
178 T TempValue{};
179 if (!Converter.ConfigValueFromString(Config, InStringValue, TempValue))
180 {
181 return false;
182 }
183
184 SetValue(MoveTemp(TempValue));
185 return true;
186 }
187
191 bool SetValueFromStringArray(const TArray<FString>& InStringArray)
192 {
193 TOnlineSubsystemAccelByteConfigArrayValueConverter<T> Converter{};
194
195 T TempValue{};
196 if (!Converter.template ConfigArrayValueFromStringArray<T>(Config, InStringArray, TempValue))
197 {
198 return false;
199 }
200
201 SetValue(MoveTemp(TempValue));
202 return true;
203 }
204
208 const T& GetValue() const
209 {
210 return Value;
211 }
212
216 const T& GetDefaultValue() const
217 {
218 return DefaultValue;
219 }
220
224 const bool IsDefault() const
225 {
226 return Value == DefaultValue;
227 }
228
232 void Reset()
233 {
234 Value = DefaultValue;
235 }
236
237PACKAGE_SCOPE:
242 TOnlineSubsystemAccelByteConfigEntry() = default;
243
248 void Init(TSharedRef<class FOnlineSubsystemAccelByteConfig, ESPMode::ThreadSafe> InConfig
249 , const FString& InSection
250 , const FString& InKey
251 , const T& InDefaultValue
252 , const FValidatorFunction& InValidator={})
253 {
254 Config = InConfig;
255 Section = InSection;
256 Key = InKey;
257 Value = InDefaultValue;
258 DefaultValue = InDefaultValue;
259 Validator = InValidator;
260 }
261
262private:

267
271 FString Section{};
272
276 FString Key{};
277
281 T Value{};
282
286 T DefaultValue{};

291 FValidatorFunction Validator{};
