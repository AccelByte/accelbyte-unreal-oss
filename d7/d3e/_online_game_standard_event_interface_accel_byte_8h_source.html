127 template<typename T>
128 void SendEvent(int32 LocalUserNum, const TSharedRef<T>& Payload, const FDateTime& ClientTimestamp = FDateTime::UtcNow())
129 {
130 if (!TIsDerivedFrom<T, FAccelByteModelsGameStandardEvent>::IsDerived)
131 {
132 OnError((int32)AccelByte::ErrorCodes::InvalidRequest, TEXT("Payload must derive from FAccelByteModelsGameStandardEvent!"), LocalUserNum, TEXT(""));
133 return;
134 }
135
136 if (LocalUserNum < 0)
137 {
138 LocalUserNum = GetLocalUserNumCached();
139 }
140
141 SetDelegatesAndInterval(LocalUserNum);
142
143 FOnlineSubsystemAccelBytePtr AccelByteSubsystemPtr = AccelByteSubsystem.Pin();
144 if (!AccelByteSubsystemPtr.IsValid())
145 {
146 AB_OSS_PTR_INTERFACE_TRACE_END_VERBOSITY(Warning, TEXT("Failed, AccelbyteSubsystem is invalid"));
147 return;
148 }
149
150 const FOnlineIdentityAccelBytePtr IdentityInterface = StaticCastSharedPtr<FOnlineIdentityAccelByte>(AccelByteSubsystemPtr->GetIdentityInterface());
151 if (IdentityInterface.IsValid())
152 {
153 if (IdentityInterface->GetLoginStatus(LocalUserNum) == ELoginStatus::LoggedIn)
154 {
155 if (!IsRunningDedicatedServer())
156 {
157 const auto ApiClient = AccelByteSubsystemPtr->GetApiClient(LocalUserNum);
158 if (ApiClient.IsValid())
159 {
160 const auto GameStandardEvent = ApiClient->GetGameStandardEventApi().Pin();
161 if (GameStandardEvent.IsValid())
162 {
163 GameStandardEvent->SendGameStandardEventData(Payload, AccelByte::FVoidHandler::CreateThreadSafeSP(this, &FOnlineGameStandardEventAccelByte::OnSuccess, LocalUserNum, Payload->GetGameStandardEventName()), FErrorHandler::CreateThreadSafeSP(this, &FOnlineGameStandardEventAccelByte::OnError, LocalUserNum, Payload->GetGameStandardEventName()), ClientTimestamp);
164 }
165 }
166 }
167 else
168 {
169
170 const FAccelByteInstancePtr AccelByteInstance = GetAccelByteInstance().Pin();
171 if(AccelByteInstance.IsValid())
172 {
173 const AccelByte::FServerApiClientPtr ApiClient = AccelByteInstance->GetServerApiClient();
174 if (ApiClient.IsValid())
175 {
176 ApiClient->ServerGameStandardEvent.SendGameStandardEventData(Payload, AccelByte::FVoidHandler::CreateThreadSafeSP(this, &FOnlineGameStandardEventAccelByte::OnSuccess, LocalUserNum, Payload->GetGameStandardEventName()), FErrorHandler::CreateThreadSafeSP(this, &FOnlineGameStandardEventAccelByte::OnError, LocalUserNum, Payload->GetGameStandardEventName()), ClientTimestamp);
177 }
178 }
179 }
180 }
181 else
182 {
183 Payload->GameStandardEventName = Payload->GetGameStandardEventName();
184 ConvertAndAddToCache(LocalUserNum, Payload, Payload->GetGameStandardEventName(), ClientTimestamp);
185 }
186 }
187 }
